	group by a.cultura
	union
	SELECT i.cultura FROM agrosystem.itens i
	group by i.cultura
	union
	SELECT m.cultura FROM agrosystem.manutencoes m
	group by m.cultura
	union
	SELECT s.cultura FROM agrosystem.setoressafras s
	group by s.cultura
