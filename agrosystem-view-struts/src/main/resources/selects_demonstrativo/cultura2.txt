 SELECT
			 	td_FULL.nome nome, sum(td_FULL.custoTotal) custoTotal 
			 FROM 
			 (
			 /* Total dos valores aplicados na Cultura informada*/
			 SELECT
			 	t.nome nome, sum(custoTotal) custoTotal
			 FROM
			 	sistema_aplicacoes as a,
			 	sistema_produtos as p,
			 	sistema_tipos as t
			 WHERE
				a.idSafra = :idSafraParam
				and a.idCultura = :idCulturaParam
				AND t.classificacao = :classificacaoParam
				and a.tipo = 'Cultura' /*FIXO*/	
			 	and a.idProduto = p.id
			 	AND t.id = p.idTipo			 			 	
			 group by t.nome


			 UNION ALL

			/* Total dos valores aplicados nos Setores com a Cultura informada*/
			 SELECT
			 	t.nome nome, sum(custoTotal) custoTotal
			 FROM
			 	sistema_aplicacoes as a,
			 	sistema_produtos as p,
			 	sistema_tipos as t,
			 	sistema_setoressafra as ss,
			 	sistema_variedades v
			 WHERE
				ss.idSafra = :idSafraParam				
				and v.idCultura = :idCulturaParam
				AND t.classificacao = :classificacaoParam
				and a.tipo = 'Safra/Setor' /*FIXO*/
			 	and a.idProduto = p.id				
			 	AND t.id = p.idTipo
			 	and v.id = ss.idVariedade
			 	AND a.idSafra = ss.idSafra
			 	AND a.idSetor = ss.idSetor			 	
			 group by t.nome


			 UNION ALL


			 SELECT
			   tb_totalAplicadoTudo.nome nome, (tb_totalAreaSetores.totalAreaSetores/tb_areaSetor.areaSetor)* tb_totalAplicadoTudo.totalAplicadoCultura as custoTotal
			 FROM
			   /* Total dos valores aplicados para TUDO*/
			 	(SELECT
					t.nome nome, sum(aplicacao.custoTotal) totalAplicadoCultura
			 	FROM
					sistema_aplicacoes aplicacao,
					sistema_produtos as p,
			 		sistema_tipos t
			 	WHERE
					aplicacao.tipo = 'Tudo'
					AND t.classificacao = :classificacaoParam
					and aplicacao.idSafra = :idSafraParam
					and aplicacao.idProduto = p.id
			 		AND t.id = p.idTipo
					group by t.nome) as tb_totalAplicadoTudo,
			   /* Tamanho total dos SETORES que foram plantadas a CULTURA informada*/
			 	(SELECT
					sum(safrasetor.area) totalAreaSetores
			 	FROM
					sistema_setoressafra safrasetor,
			 		sistema_setores setor,
			 		sistema_safras safra,
			 		sistema_variedades variedade
			 	WHERE
					safrasetor.idSetor = setor.id
					AND safrasetor.idSafra = safra.id
			 		AND safrasetor.idVariedade = variedade.id
			 		and safrasetor.idSafra = :idSafraParam
					AND variedade.idCultura = :idCulturaParam) as tb_totalAreaSetores,
			   /* Tamanho total da area da Safra informada*/
			 	(SELECT
					sum(safrasetor.area) areaSetor
			 	FROM
					sistema_setoressafra safrasetor,
			 		sistema_setores setor,
			 		sistema_safras safra,
			 		sistema_variedades variedade
			 	WHERE
			 	  safrasetor.idSetor = setor.id
			 	  AND safrasetor.idSafra = safra.id
			 		AND safrasetor.idVariedade = variedade.id
			 		and safrasetor.idSafra = :idSafraParam) as tb_areaSetor
			 ) as td_FULL
			 group by td_FULL.nome
